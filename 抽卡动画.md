一、整体结构
我们将创建一个 GachaAnimation.tsx 组件，它接受一个 onComplete 回调，在动画结束后通知抽卡结果。

tsx
// components/GachaAnimation.tsx
import { useRef, useEffect } from 'react';
import { useFrame, useThree } from '@react-three/fiber';
import * as THREE from 'three';

export const GachaAnimation = ({ onComplete }: { onComplete: () => void }) => {
  const groupRef = useRef<THREE.Group>(null);
  const ribbonRef = useRef<THREE.Mesh>(null);
  const particlesRef = useRef<THREE.Points>(null);
  const zipperPointsRef = useRef<THREE.Points>(null);
  const startTimeRef = useRef(performance.now());
  const openedRef = useRef(false);

  // 1. 程序化生成“礼包”模型
  useEffect(() => {
    if (!groupRef.current) return;

    // 主体：圆角立方体 (通过 BoxGeometry + 顶点偏移实现简单圆角)
    const bodyGeo = new THREE.BoxGeometry(1.8, 1.8, 1.8, 16, 16, 16);
    const positions = bodyGeo.attributes.position.array;
    for (let i = 0; i < positions.length; i += 3) {
      let x = positions[i];
      let y = positions[i+1];
      let z = positions[i+2];
      // 圆角效果：将边缘顶点向内收缩
      const max = 0.9;
      if (Math.abs(x) > max) x = Math.sign(x) * max;
      if (Math.abs(y) > max) y = Math.sign(y) * max;
      if (Math.abs(z) > max) z = Math.sign(z) * max;
      positions[i] = x;
      positions[i+1] = y;
      positions[i+2] = z;
    }
    bodyGeo.computeVertexNormals();
    const bodyMat = new THREE.MeshStandardMaterial({ color: 0xe83e8c, roughness: 0.3, metalness: 0.1 });
    const body = new THREE.Mesh(bodyGeo, bodyMat);
    body.castShadow = true;
    body.receiveShadow = false;
    groupRef.current.add(body);

    // 丝带 (横向)
    const ribbonGeo = new THREE.BoxGeometry(2.0, 0.1, 0.4);
    const ribbonMat = new THREE.MeshStandardMaterial({ color: 0xffd966, emissive: 0x442200 });
    const ribbonH = new THREE.Mesh(ribbonGeo, ribbonMat);
    ribbonH.position.y = 0.6;
    groupRef.current.add(ribbonH);
    const ribbonV = new THREE.Mesh(ribbonGeo, ribbonMat);
    ribbonV.position.x = 0.6;
    ribbonV.rotation.z = Math.PI / 2;
    groupRef.current.add(ribbonV);

    // 蝴蝶结 (简单两个环)
    const bowGeo = new THREE.TorusGeometry(0.2, 0.05, 16, 32);
    const bowMat = new THREE.MeshStandardMaterial({ color: 0xffaa66 });
    const bowLeft = new THREE.Mesh(bowGeo, bowMat);
    bowLeft.position.set(-0.5, 0.9, 0.5);
    bowLeft.rotation.z = 0.5;
    groupRef.current.add(bowLeft);
    const bowRight = new THREE.Mesh(bowGeo, bowMat);
    bowRight.position.set(0.5, 0.9, 0.5);
    bowRight.rotation.z = -0.5;
    groupRef.current.add(bowRight);
  }, []);

  // 2. 粒子系统 (拉链光点 + 爆炸粒子)
  useEffect(() => {
    if (!groupRef.current) return;
    const scene = groupRef.current.parent;
    if (!scene) return;

    // 拉链光点：沿礼包正面中线分布的粒子
    const zipperCount = 30;
    const zipperPositions = [];
    for (let i = 0; i <= zipperCount; i++) {
      const t = i / zipperCount; // 0 底部 -> 1 顶部
      const y = -0.8 + t * 1.6;
      zipperPositions.push(0.91, y, 0.91); // 正面边缘
    }
    const zipperGeo = new THREE.BufferGeometry();
    zipperGeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array(zipperPositions), 3));
    const zipperMat = new THREE.PointsMaterial({ color: 0xffaa44, size: 0.05, transparent: true, blending: THREE.AdditiveBlending });
    const zipperPoints = new THREE.Points(zipperGeo, zipperMat);
    groupRef.current.add(zipperPoints);
    zipperPointsRef.current = zipperPoints;

    // 爆炸粒子 (初始不可见)
    const particleCount = 800;
    const particlePositions = new Float32Array(particleCount * 3);
    for (let i = 0; i < particleCount; i++) {
      // 随机分布在礼包内部
      particlePositions[i*3] = (Math.random() - 0.5) * 1.2;
      particlePositions[i*3+1] = (Math.random() - 0.5) * 1.2;
      particlePositions[i*3+2] = (Math.random() - 0.5) * 1.2;
    }
    const particleGeo = new THREE.BufferGeometry();
    particleGeo.setAttribute('position', new THREE.BufferAttribute(particlePositions, 3));
    // 速度属性 (将在动画中更新)
    const velocities = new Array(particleCount).fill(0).map(() => new THREE.Vector3(
      (Math.random() - 0.5) * 0.5,
      (Math.random() - 0.5) * 0.5 + 0.5,
      (Math.random() - 0.5) * 0.5
    ));
    const particleMat = new THREE.PointsMaterial({ color: 0xff66cc, size: 0.05, transparent: true, blending: THREE.AdditiveBlending });
    const particles = new THREE.Points(particleGeo, particleMat);
    particles.visible = false;
    groupRef.current.add(particles);
    particlesRef.current = particles;

    // 存储速度以便动画
    (particles.userData as any).velocities = velocities;
  }, []);

  // 3. 动画循环：拉开 + 光点游走 + 爆炸
  useFrame((_, delta) => {
    if (!groupRef.current) return;
    const elapsed = (performance.now() - startTimeRef.current) / 1000;
    const duration = 2.0; // 总动画时长2秒

    if (elapsed >= duration && !openedRef.current) {
      openedRef.current = true;
      if (particlesRef.current) particlesRef.current.visible = true;
      // 延迟回调，让粒子效果播放一会儿
      setTimeout(() => {
        onComplete();
      }, 500);
    }

    // 阶段1 (0~1.2秒): 礼包打开 (scale.y 逐渐缩小并偏移开口)
    const openProgress = Math.min(1.0, elapsed / 1.2);
    if (openProgress < 1.0) {
      // 用 easeOutBack 曲线让打开更有弹性
      const easeOutBack = 1 - Math.pow(1 - openProgress, 2);
      // 礼包主体裂开：把上半部分向上移动
      // 实际实现中可以通过修改子物体位置，这里简化：让整个礼包 scale.y 减小并向上移动开口边沿
      groupRef.current.children.forEach(child => {
        if (child.isMesh && child !== particlesRef.current && child !== zipperPointsRef.current) {
          // 将 y > 0 的顶点向上偏移（需要在顶点着色器中做，但为了简化，我们用整体缩放模拟）
          // 更精细的实现需要修改几何体属性，这里用缩放模拟打开效果
          if (child.position.y > 0) {
            child.position.y += delta * 2.0;
          }
        }
      });
    }

    // 阶段2 (0~1.5秒): 拉链光点依次亮起
    if (zipperPointsRef.current && elapsed < 1.5) {
      const progress = elapsed / 1.5;
      const geometry = zipperPointsRef.current.geometry;
      const positions = geometry.attributes.position.array;
      // 根据进度改变粒子大小或颜色（通过材质）
      const color = new THREE.Color().setHSL(0.12, 1.0, 0.5 + progress * 0.3);
      (zipperPointsRef.current.material as THREE.PointsMaterial).color = color;
      // 也可以动态改变透明度
      (zipperPointsRef.current.material as THREE.PointsMaterial).opacity = Math.min(1.0, progress * 1.5);
    }

    // 阶段3: 爆炸粒子 (打开后)
    if (openedRef.current && particlesRef.current && particlesRef.current.visible) {
      const particleGeo = particlesRef.current.geometry;
      const positions = particleGeo.attributes.position.array;
      const velocities = (particlesRef.current.userData as any).velocities;
      let allDead = true;
      for (let i = 0; i < velocities.length; i++) {
        // 更新位置
        positions[i*3] += velocities[i].x * delta * 3;
        positions[i*3+1] += velocities[i].y * delta * 3;
        positions[i*3+2] += velocities[i].z * delta * 3;
        // 重力影响
        velocities[i].y -= delta * 2;
        // 超出范围标记为死亡（不显示），简单起见我们一直更新
        if (Math.abs(positions[i*3]) < 5 && positions[i*3+1] > -2) allDead = false;
      }
      particleGeo.attributes.position.needsUpdate = true;
      // 逐渐淡出
      const mat = particlesRef.current.material as THREE.PointsMaterial;
      mat.opacity -= delta * 2;
      if (mat.opacity <= 0) {
        particlesRef.current.visible = false;
      }
    }
  });

  return (
    <group ref={groupRef} position={[0, 0, 0]}>
      {/* 这里添加一些环境光或自发光效果 */}
      <pointLight position={[1, 2, 1]} intensity={0.5} color="#ffaa88" />
    </group>
  );
};
二、在抽卡页面中使用
tsx
// ui/pages/GachaPage.tsx
import { useState } from 'react';
import { GachaAnimation } from '../../components/GachaAnimation';

export const GachaPage = () => {
  const [showAnimation, setShowAnimation] = useState(false);
  const [result, setResult] = useState<string | null>(null);

  const startGacha = () => {
    setShowAnimation(true);
  };

  const handleAnimationComplete = () => {
    // 随机抽卡结果
    const items = ['★3 角色', '★4 角色', '★5 角色'];
    const picked = items[Math.floor(Math.random() * items.length)];
    setResult(picked);
    setShowAnimation(false);
  };

  return (
    <div className="gacha-page">
      <button onClick={startGacha}>进行一次抽取</button>
      {showAnimation && (
        <div className="animation-overlay">
          <GachaAnimation onComplete={handleAnimationComplete} />
        </div>
      )}
      {result && <div className="result">获得：{result}</div>}
    </div>
  );
};
