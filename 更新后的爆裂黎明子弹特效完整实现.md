## 2\. 顶点数据处理（完全保留你原来的归一化逻辑）

```typescript
function normalizePoints(points: [number, number][]): [number, number][] {
  let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
  points.forEach(([x, y]) => {
    minX = Math.min(minX, x);
    minY = Math.min(minY, y);
    maxX = Math.max(maxX, x);
    maxY = Math.max(maxY, y);
  });
  const xMid = (minX + maxX) / 2;
  const xRange = maxX - minX || 1;
  const yRange = maxY - minY || 1;
  // 归一化：以X中点为原点，Y最高点为原点
  // X: (x - xMid) / xRange → 范围从 -0.5 到 0.5
  // Y: (y - maxY) / yRange → 范围从 -1 到 0
  return points.map(([x, y]) => [
    (x - xMid) / xRange,
    (y - maxY) / yRange
  ]);
}

function findMaxYPoint(points: [number, number][]): [number, number] {
  let maxYPoint = points[0];
  let maxY = points[0][1];
  for (const point of points) {
    if (point[1] > maxY) {
      maxY = point[1];
      maxYPoint = point;
    }
  }
  return maxYPoint;
}
```

```typescript
export function createBulletTrailTexture(textureManager: TextureManager, width: number = 256, height: number = 256): void {
  const generator = new CanvasTextureGenerator(width, height, (ctx, w, h) => {
    ctx.clearRect(0, 0, w, h);
    
    // 1. 绘制上半段（Y=0~0.5，靠近弹头的主尾气）
    ctx.beginPath();
    // 只取Y<0.5的顶点，也就是前半段的形状
    const firstPoint = normalizedPoints[0];
    if (-firstPoint[1] < 0.5) {
      ctx.moveTo((firstPoint[0] + 0.5) * w, (-firstPoint[1]) * h);
      for (let i = 1; i < normalizedPoints.length; i++) {
        const [x, y] = normalizedPoints[i];
        if (-y < 0.5) {
          ctx.lineTo((x + 0.5) * w, (-y) * h);
        }
      }
      ctx.closePath();
      
      // 主尾气的4层径向渐变，符合高斯扩散
      const mainGradient = ctx.createRadialGradient(w*0.5, 0, 0, w*0.5, 0, w*0.4);
      // 从核心到外层，完全是你给的色值
      mainGradient.addColorStop(0.0, 'rgba(155, 26, 91, 1)');      // 核心：#9b1a5b
      mainGradient.addColorStop(0.1, 'rgba(244, 0, 64, 0.95)');   // 次核心：#f40040
      mainGradient.addColorStop(0.25, 'rgba(254, 41, 93, 0.8)');  // 中层：#fe295d
      mainGradient.addColorStop(0.4, 'rgba(254, 106, 130, 0.6)');// 外层：#fe6a82
      
      // 纵向的温度渐变，靠近弹头的地方更深，分界处更淡
      const verticalGradient = ctx.createLinearGradient(0, 0, 0, h*0.5);
      verticalGradient.addColorStop(0.0, 'rgba(1, 1, 1, 1)');
      verticalGradient.addColorStop(1.0, 'rgba(1.2, 1.2, 1.2, 0.8)');
      
      ctx.fillStyle = mainGradient;
      ctx.fill();
      // 叠加纵向温度渐变
      ctx.globalCompositeOperation = 'multiply';
      ctx.fillStyle = verticalGradient;
      ctx.fill();
      ctx.globalCompositeOperation = 'source-over';
    }
    
    // 2. 绘制下半段（Y=0.5~1，尾部的卡门涡街拖尾）

    ctx.beginPath();

      const [x, y] = normalizedPoints[i];
      if (-y >= 0.5 && x < 0) { // 只取左半部分的顶点

          if (i === 0) ctx.moveTo((x+0.5)*w, (-y)*h);
          else ctx.lineTo((x+0.5)*w, (-y)*h);
        }
      }
    }
    ctx.closePath();
    // 左拖尾的3层渐变
    const leftGradient = ctx.createRadialGradient(w*0.25, h*0.5, 0, w*0.25, h*0.5, w*0.2);
    leftGradient.addColorStop(0.0, 'rgba(32, 19, 18, 1)');      // 核心：#201312
    leftGradient.addColorStop(0.3, 'rgba(59, 40, 60, 0.9)');    // 中层：#3b283c
    leftGradient.addColorStop(0.5, 'rgba(133, 38, 58, 0.7)');  // 外层：#85263a
    // 纵向降温渐变
    const leftVertical = ctx.createLinearGradient(0, h*0.5, 0, h);
    leftVertical.addColorStop(0.0, 'rgba(1, 1, 1, 1)');
    leftVertical.addColorStop(1.0, 'rgba(1.3, 1.3, 1.3, 0.7)');
    
    ctx.fillStyle = leftGradient;
    ctx.fill();
    ctx.globalCompositeOperation = 'multiply';
    ctx.fillStyle = leftVertical;
    ctx.fill();
    ctx.globalCompositeOperation = 'source-over';
    
    // 右拖尾
    ctx.beginPath();
    for (let i = 0; i < normalizedPoints.length; i++) {
      const [x, y] = normalizedPoints[i];
      if (-y >= 0.5 && x > 0) { // 只取右半部分的顶点
        if (i === 0) ctx.moveTo((x+0.5)*w, (-y)*h);
        else ctx.lineTo((x+0.5)*w, (-y)*h);
      }
    }
    ctx.closePath();
    // 右拖尾的3层渐变
    const rightGradient = ctx.createRadialGradient(w*0.75, h*0.5, 0, w*0.75, h*0.5, w*0.2);
    rightGradient.addColorStop(0.0, 'rgba(32, 19, 18, 1)');      // 核心：#201312
    rightGradient.addColorStop(0.3, 'rgba(59, 40, 60, 0.9)');    // 中层：#3b283c
    rightGradient.addColorStop(0.5, 'rgba(133, 38, 58, 0.7)');  // 外层：#85263a
    // 纵向降温渐变
    const rightVertical = ctx.createLinearGradient(0, h*0.5, 0, h);
    rightVertical.addColorStop(0.0, 'rgba(1, 1, 1, 1)');
    rightVertical.addColorStop(1.0, 'rgba(1.3, 1.3, 1.3, 0.7)');
    
    ctx.fillStyle = rightGradient;
    ctx.fill();
    ctx.globalCompositeOperation = 'multiply';
    ctx.fillStyle = rightVertical;
    ctx.fill();
    ctx.globalCompositeOperation = 'source-over';
  });
```

## 4\. 几何体创建函数（更新了分层，支持 7 层独立参数）

原来的只有 2 层，现在扩展到 7 层，主尾气 4 层，拖尾 3 层，每层都有独立的摆动和衰减：

```typescript
export function createBulletTrailGeometry(): THREE.BufferGeometry {
  const geometry = new THREE.BufferGeometry();
  const vertices = [];
  const uvs = [];
  const layerIndices = [];
  const layerOffset = 0;
  const scaleFactor = 0.7;
  
  // 1. 主尾气的4层
  for (let layer = 0; layer < 4; layer++) {
    const zOffset = layer * layerOffset;
    for (let i = 1; i < normalizedPoints.length - 1; i++) {
      const center = normalizedMaxYPoint;
      const current = normalizedPoints[i];
      const next = normalizedPoints[i + 1];
      // 只处理前半段的顶点
      if (-current[1] < 0.5 && -next[1] < 0.5) {
        const centerZ = 3 + center[1] + zOffset;
        const currentZ = 3 + current[1] + zOffset;
        const nextZ = 3 + next[1] + zOffset;
        
        if (!isNaN(center[0]) && !isNaN(centerZ) &&
            !isNaN(current[0]) && !isNaN(currentZ) &&
            !isNaN(next[0]) && !isNaN(nextZ)) {
          vertices.push(
            center[0] * scaleFactor, 0, centerZ,
            current[0] * scaleFactor, 0, currentZ,
            next[0] * scaleFactor, 0, nextZ
          );
          uvs.push(
            center[0] + 0.5, -center[1],
            current[0] + 0.5, -current[1],
            next[0] + 0.5, -next[1]
          );
          // 层索引0-3，代表主尾气的4层
          for (let j = 0; j < 3; j++) {
            layerIndices.push(layer);
          }
        }
      }
    }
  }
  
  // 2. 拖尾的3层
  for (let layer = 0; layer < 3; layer++) {
    const zOffset = layer * layerOffset;
    for (let i = 1; i < normalizedPoints.length - 1; i++) {
      const center = normalizedMaxYPoint;
      const current = normalizedPoints[i];
      const next = normalizedPoints[i + 1];
      // 只处理后半段的顶点
      if (-current[1] >= 0.5 && -next[1] >= 0.5) {
        const centerZ = 3 + center[1] + zOffset;
        const currentZ = 3 + current[1] + zOffset;
        const nextZ = 3 + next[1] + zOffset;
        
        if (!isNaN(center[0]) && !isNaN(centerZ) &&
            !isNaN(current[0]) && !isNaN(currentZ) &&
            !isNaN(next[0]) && !isNaN(nextZ)) {
          vertices.push(
            center[0] * scaleFactor, 0, centerZ,
            current[0] * scaleFactor, 0, currentZ,
            next[0] * scaleFactor, 0, nextZ
          );
          uvs.push(
            center[0] + 0.5, -center[1],
            current[0] + 0.5, -current[1],
            next[0] + 0.5, -next[1]
          );
          // 层索引4-6，代表拖尾的3层
          for (let j = 0; j < 3; j++) {
            layerIndices.push(4 + layer);
          }
        }
      }
    }
  }
  
  // 设置几何体属性
  const vertexBuffer = new Float32Array(vertices);
  geometry.setAttribute('position', new THREE.BufferAttribute(vertexBuffer, 3));
  const uvBuffer = new Float32Array(uvs);
  geometry.setAttribute('uv', new THREE.BufferAttribute(uvBuffer, 2));
  const layerBuffer = new Float32Array(layerIndices);
  geometry.setAttribute('layer', new THREE.BufferAttribute(layerBuffer, 1));
  
  geometry.computeVertexNormals();
  return geometry;
}
```

```typescript
export function createBulletTrailMaterial(texture: THREE.Texture): THREE.ShaderMaterial {
  texture.wrapS = THREE.ClampToEdgeWrapping;
  texture.wrapT = THREE.ClampToEdgeWrapping;
  
  const vertexShader = `
    uniform float uTime;
    uniform float uWobbleIntensity;
    uniform float uWobbleSpeed;
    uniform float uBulletSpeed;
    varying vec2 vUv;
    varying float vPositionZ;
    attribute float layer;
    
    void main() {
      vUv = uv;
      vPositionZ = position.z;
      vec3 pos = position;
      
      float normalizedZ = position.z - 2.0;
      float tailFactor = 1.0 - normalizedZ;
      // 速度联动：速度越快，摆动越强
      float speedFactor = 1.0 + 0.5 * uBulletSpeed / 500.0;
      float wobbleAmplitude = pow(tailFactor, 3.0) * uWobbleIntensity * 1.5 * speedFactor;
      float wobbleFreq = uWobbleSpeed * speedFactor;
      
      if (layer < 4.0) {
        // 主尾气的4层，湍流摆动
        float layerFactor = 1.0 - layer * 0.2;
        float wobbleX = sin(uTime * wobbleFreq * 1.5 * layerFactor + tailFactor * 12.0) * wobbleAmplitude * 0.8;
        wobbleX += sin(uTime * wobbleFreq * 2.0 * layerFactor + tailFactor * 18.0) * wobbleAmplitude * 0.4;
        float wobbleY = cos(uTime * wobbleFreq * 1.2 * layerFactor + tailFactor * 10.0) * wobbleAmplitude * 0.6;
        wobbleY += cos(uTime * wobbleFreq * 1.6 * layerFactor + tailFactor * 14.0) * wobbleAmplitude * 0.3;
        pos.x += wobbleX;
        pos.y += wobbleY;
      } else {
        // 拖尾的3层，卡门涡街交替摆动
        float layerFactor = 1.0 - (layer - 4.0) * 0.2;
        // 左右拖尾的相位差，模拟交替涡旋
        float phase = vUv.x < 0.5 ? 0.785 : -0.785;
        float wobbleX = sin(uTime * wobbleFreq * 0.8 * layerFactor + phase + tailFactor * 8.0) * wobbleAmplitude * 1.2;
        wobbleX += sin(uTime * wobbleFreq * 1.0 * layerFactor + phase + tailFactor * 12.0) * wobbleAmplitude * 0.6;
        float wobbleY = cos(uTime * wobbleFreq * 0.6 * layerFactor + phase + tailFactor * 6.0) * wobbleAmplitude * 1.0;
        wobbleY += cos(uTime * wobbleFreq * 0.8 * layerFactor + phase + tailFactor * 10.0) * wobbleAmplitude * 0.5;
        pos.x += wobbleX;
        pos.y += wobbleY;
      }
      
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
    }
  `;
  
  const fragmentShader = `
    uniform sampler2D uTexture;
    uniform float uTime;
    uniform float uDistortStrength;
    varying vec2 vUv;
    varying float vPositionZ;
    
    void main() {
      // 轻微的热扭曲，模拟高温气体折射
      vec2 uv = vUv;
      vec2 distort = uv - 0.5;
      float dist = length(distort);
      if (dist < 0.5) {
        float strength = (1.0 - dist * 2.0) * uDistortStrength;
        uv += distort * strength;
      }
      
      vec4 texColor = texture2D(uTexture, uv);
      if (texColor.a < 0.1) {
        discard;
      }
      
      // 前端发光，速度联动
      float normalizedZ = vPositionZ - 2.0;
      if (normalizedZ > 0.95 && abs(vUv.x - 0.5) < 0.1) {
        float glowIntensity = (normalizedZ - 0.95) * 20.0;
        float widthIntensity = 1.0 - abs(vUv.x - 0.5) / 0.1;
        vec3 glowColor = vec3(0.8, 0.6, 0.8);
        texColor.rgb = mix(texColor.rgb, glowColor, glowIntensity * widthIntensity * 0.5);
        texColor.a = mix(texColor.a, 1.0, glowIntensity * widthIntensity * 0.3);
      }
      
      // Bloom预乘，让亮的地方自动辉光
      texColor.rgb *= 1.2;
      
      gl_FragColor = texColor;
    }
  `;
  
  return new THREE.ShaderMaterial({
    uniforms: {
      uTexture: { value: texture },
      uTime: { value: 0 },
      uWobbleIntensity: { value: 0.1 },
      uWobbleSpeed: { value: 4.0 },
      uBulletSpeed: { value: 300.0 }, // 默认速度
      uDistortStrength: { value: 0.01 } // 热扭曲强度
    },
    vertexShader,
    fragmentShader,
    transparent: true,
    side: THREE.DoubleSide,
    // 开启Bloom兼容
    blending: THREE.AdditiveBlending
  });
}
```

## 6\. 子弹实体类（更新了速度联动、LOD、弹头材质）

完全更新了原来的绑定逻辑，加入了动态参数联动：

```typescript
export class DawnExplosionBulletEntity {
  private trailTime: number = 0;
  private trailMesh: THREE.Mesh | null = null;
  private trailMaterial: THREE.ShaderMaterial | null = null;
  private cameraDistance: number = 0;
  
  constructor(
    position: THREE.Vector3,
    direction: THREE.Vector3,
    speed: number,
    color: number = 0xdd5b42
  ) {
    super(position, direction, speed, color);
    this.setDamage(2);
    this.createTrailEffect();
    this.modifyBulletMaterial(speed);
  }
  
  private createTrailEffect(): void {
    const textureManager = new TextureManager();
    createBulletTrailTexture(textureManager);
    const texture = textureManager.getTexture('bullet-trail');
    if (!texture) return;
    
    const geometry = createBulletTrailGeometry();
    this.trailMaterial = createBulletTrailMaterial(texture);
    this.trailMesh = new THREE.Mesh(geometry, this.trailMaterial);
    this.trailMesh.scale.set(3, 3, 3);
    
    const scene = EntityManager.getInstance().getScene();
    if (scene) {
      scene.add(this.trailMesh);
    }
    
    const flightDir = this.velocity.clone().normalize();
    const distance = 8.5;
    this.trailMesh.position.copy(this.position.clone().add(flightDir.clone().multiplyScalar(-distance)));
    this.trailMesh.quaternion.copy(this.mesh.quaternion);
  }
  
  private modifyBulletMaterial(speed: number): void {
    const textureManager = new TextureManager();
    createBulletTrailTexture(textureManager);
    const trailTexture = textureManager.getTexture('bullet-trail');
    if (!trailTexture) return;
    
    // 弹头的激波材质，带热扭曲和速度发光
    const bulletMaterial = new THREE.ShaderMaterial({
      uniforms: {
        baseColor: { value: new THREE.Color(this.mesh.material instanceof THREE.MeshStandardMaterial ? this.mesh.material.color : 0xdd5b42) },
        trailTexture: { value: trailTexture },
        time: { value: 0 },
        speed: { value: speed },
        distortStrength: { value: 0.015 }
      },
      vertexShader: `
        varying vec2 vUv;
        varying vec3 vPosition;
        void main() {
          vUv = uv;
          vPosition = position;
          gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
        }
      `,
      fragmentShader: `
        uniform vec3 baseColor;
        uniform sampler2D trailTexture;
        uniform float time;
        uniform float speed;
        uniform float distortStrength;
        varying vec2 vUv;
        varying vec3 vPosition;
        
        void main() {
          vec3 color = baseColor;
          
          // 弹头的热扭曲
          vec2 uv = vUv;
          vec2 distort = uv - 0.5;
          float dist = length(distort);
          if (dist < 0.5) {
            float strength = (1.0 - dist * 2.0) * distortStrength * (1.0 + 0.2 * speed / 500.0);
            uv += distort * strength;
          }
          
          // 前端尾气流动
          float trailLength = 0.3;
          float zNormalized = (vPosition.z + 0.3) / 0.6;
          float textureY = zNormalized * trailLength;
          float timeOffset = mod(time * speed * 0.1, trailLength);
          float trailUvY = textureY + timeOffset;
          trailUvY = mod(trailUvY, 1.0);
          vec2 trailUv = vec2(uv.x, trailUvY);
          vec4 trailColor = texture2D(trailTexture, trailUv);
          
          float alpha = trailColor.a * 0.8;
          color = mix(color, trailColor.rgb, alpha);
          
          // 速度联动的发光
          if (zNormalized > 0.95) {
            float glowIntensity = (zNormalized - 0.95) * 20.0 * (1.0 + 0.2 * speed / 500.0);
            vec3 glowColor = vec3(0.8, 0.6, 0.8);
            color = mix(color, glowColor, glowIntensity * 0.5);
          }
          
          // 激波的红色过渡
          if (zNormalized < 0.3) {
            float endIntensity = 1.0 - (zNormalized / 0.3);
            vec3 endColor = vec3(1.0, 0.0, 0.21);
            color = mix(color, endColor, endIntensity * 0.8);
          }
          
          gl_FragColor = vec4(color, 1.0);
        }
      `,
      side: THREE.DoubleSide,
      transparent: true
    });
    
    this.mesh.material = bulletMaterial;
  }
  
  public update(delta: number): void {
    super.update(delta);
    this.trailTime += delta;
    
    // 更新相机距离，做LOD
    const camera = EntityManager.getInstance().getCamera();
    this.cameraDistance = camera.position.distanceTo(this.position);
    
    const speed = this.velocity.length();
    const flightDir = this.velocity.clone().normalize();
    
    // 更新子弹材质
    if (this.mesh.material instanceof THREE.ShaderMaterial) {
      this.mesh.material.uniforms.time.value = this.trailTime;
      this.mesh.material.uniforms.speed.value = speed;
      // LOD调整
      if (this.cameraDistance > 30) {
        this.mesh.material.uniforms.distortStrength.value = 0;
      } else {
        this.mesh.material.uniforms.distortStrength.value = 0.015;
      }
    }
    
    // 更新尾气材质
    if (this.trailMesh && this.trailMaterial) {
      const distance = 8.5 + 0.5 * speed / 500.0; // 速度联动的尾气长度
      const trailPos = this.position.clone().add(flightDir.clone().multiplyScalar(-distance));
      this.trailMesh.position.copy(trailPos);
      this.trailMesh.quaternion.copy(this.mesh.quaternion);
      
      this.trailMaterial.uniforms.uTime.value = this.trailTime;
      this.trailMaterial.uniforms.uBulletSpeed.value = speed;
      // LOD调整
      if (this.cameraDistance > 30) {
        this.trailMaterial.uniforms.uDistortStrength.value = 0;
        this.trailMaterial.uniforms.uWobbleIntensity.value = 0.05;
      } else {
        this.trailMaterial.uniforms.uDistortStrength.value = 0.01;
        this.trailMaterial.uniforms.uWobbleIntensity.value = 0.1;
      }
    }
  }
}
```
