1. 动画片段 AnimationClip
typescript
// src/core/AnimationClip.ts
import * as THREE from 'three';

export class AnimationClip {
  public frames: THREE.Texture[] = [];
  public frameRate: number = 12;
  public loop: boolean = true;
  private currentFrame: number = 0;
  private timeAcc: number = 0;

  constructor(frames: THREE.Texture[], frameRate: number = 12, loop: boolean = true) {
    this.frames = frames;
    this.frameRate = frameRate;
    this.loop = loop;
  }

  public update(delta: number): THREE.Texture | null {
    if (this.frames.length === 0) return null;
    this.timeAcc += delta;
    const frameInterval = 1 / this.frameRate;
    if (this.timeAcc >= frameInterval) {
      const steps = Math.floor(this.timeAcc / frameInterval);
      this.currentFrame += steps;
      this.timeAcc -= steps * frameInterval;
      if (this.currentFrame >= this.frames.length) {
        if (this.loop) {
          this.currentFrame %= this.frames.length;
        } else {
          this.currentFrame = this.frames.length - 1;
          return this.frames[this.currentFrame];
        }
      }
    }
    return this.frames[this.currentFrame];
  }

  public reset(): void {
    this.currentFrame = 0;
    this.timeAcc = 0;
  }
}
2. 纸片人动画器 PaperAnimator
typescript
// src/core/PaperAnimator.ts
import * as THREE from 'three';
import { AnimationClip } from './AnimationClip';

export class PaperAnimator {
  private frontClip: AnimationClip | null = null;
  private backClip: AnimationClip | null = null;
  private currentClip: AnimationClip | null = null;
  private material: THREE.MeshStandardMaterial;

  constructor(material: THREE.MeshStandardMaterial) {
    this.material = material;
  }

  public setFrontClip(clip: AnimationClip): void {
    this.frontClip = clip;
  }

  public setBackClip(clip: AnimationClip): void {
    this.backClip = clip;
  }

  /**
   * 根据角色位置和相机位置决定播放正面还是背面动画
   * @param charPos 角色世界坐标
   * @param cameraPos 相机世界坐标
   */
  public updateDirection(charPos: THREE.Vector3, cameraPos: THREE.Vector3): void {
    const toCamera = new THREE.Vector3().subVectors(cameraPos, charPos);
    toCamera.y = 0;
    if (toCamera.length() < 0.001) return;
    toCamera.normalize();
    const forward = new THREE.Vector3(0, 0, 1);
    const isFront = forward.dot(toCamera) > 0;
    const targetClip = isFront ? this.frontClip : this.backClip;
    if (targetClip === this.currentClip) return;
    this.currentClip = targetClip;
    this.currentClip?.reset();
  }

  public update(delta: number): void {
    if (!this.currentClip) return;
    const texture = this.currentClip.update(delta);
    if (texture) this.material.map = texture;
  }

  public setFrameRate(frontRate?: number, backRate?: number): void {
    if (frontRate !== undefined && this.frontClip) this.frontClip.frameRate = frontRate;
    if (backRate !== undefined && this.backClip) this.backClip.frameRate = backRate;
  }
}
3. 动画加载器 AnimationLoader
typescript
// src/core/AnimationLoader.ts
import * as THREE from 'three';
import { AnimationClip } from './AnimationClip';

export class AnimationLoader {
  /**
   * 加载序列帧动画
   * @param basePath 文件夹路径，如 '/textures/characters/player/front'
   * @param frameCount 帧数
   * @param startIndex 起始索引，默认1
   * @param frameRate 帧率
   * @param loop 是否循环
   */
  public static async loadClip(
    basePath: string,
    frameCount: number,
    startIndex: number = 1,
    frameRate: number = 12,
    loop: boolean = true
  ): Promise<AnimationClip> {
    const textures: THREE.Texture[] = [];
    const loader = new THREE.TextureLoader();
    const promises: Promise<THREE.Texture>[] = [];
    for (let i = 0; i < frameCount; i++) {
      const idx = startIndex + i;
      const padded = idx.toString().padStart(2, '0');
      const path = `${basePath}/frame_${padded}.png`;
      promises.push(loader.loadAsync(path));
    }
    const loaded = await Promise.all(promises);
    return new AnimationClip(loaded, frameRate, loop);
  }

  /**
   * 加载纸片人角色的正面和背面动画
   * @param baseFolder 角色根目录，应包含 front/ 和 back/ 子文件夹
   * @param frontCount 正面帧数
   * @param backCount 背面帧数
   * @param frameRate 帧率
   */
  public static async loadPaperAnimations(
    baseFolder: string,
    frontCount: number,
    backCount: number,
    frameRate: number = 12
  ): Promise<{ frontClip: AnimationClip; backClip: AnimationClip }> {
    const frontClip = await this.loadClip(`${baseFolder}/front`, frontCount, 1, frameRate, true);
    const backClip = await this.loadClip(`${baseFolder}/back`, backCount, 1, frameRate, true);
    return { frontClip, backClip };
  }
}

// src/core/Entity.ts
import * as THREE from 'three';

export abstract class Entity {
  public id: string;
  public type: string;
  public mesh: THREE.Object3D;
  public position: THREE.Vector3;
  public velocity: THREE.Vector3 = new THREE.Vector3(0, 0, 0);
  public isActive: boolean = true;
  public radius: number = 0.5;
  public health: number = 100;
  public maxHealth: number = 100;

  constructor(id: string, type: string, mesh: THREE.Object3D) {
    this.id = id;
    this.type = type;
    this.mesh = mesh;
    this.position = mesh.position.clone();
  }

  public abstract update(delta: number): void;
  public onDestroy(): void {}
  public takeDamage(amount: number): void {
    this.health = Math.max(0, this.health - amount);
    if (this.health <= 0) this.isActive = false;
  }
}
5. 角色基类 CharacterEntity（集成动画器）
typescript
// src/entities/CharacterEntity.ts
import { Entity } from '../core/Entity';
import { PaperAnimator } from '../core/PaperAnimator';
import { cameraStore } from '../core/CameraStore';
import * as THREE from 'three';

export abstract class CharacterEntity extends Entity {
  public faction: string;
  public speed: number = 5;
  public moveDirection: THREE.Vector3 = new THREE.Vector3(0, 0, 0);
  public animator: PaperAnimator | null = null;

  constructor(id: string, type: string, mesh: THREE.Object3D, faction: string = 'neutral') {
    super(id, type, mesh);
    this.faction = faction;
  }

  /**
   * 注入动画器（由子类在构造时调用）
   */
  public setAnimator(animator: PaperAnimator): void {
    this.animator = animator;
  }

  public update(delta: number): void {
    // 移动逻辑（由子类具体实现 moveDirection 的计算）
    if (this.moveDirection.length() > 0) {
      this.moveDirection.normalize();
      this.position.x += this.moveDirection.x * this.speed * delta;
      this.position.z += this.moveDirection.z * this.speed * delta;
      this.mesh.position.copy(this.position);
    }

    // 动画更新（如果有动画器）
    if (this.animator) {
      const camera = cameraStore.getCamera();
      if (camera) {
        this.animator.updateDirection(this.position, camera.position);
        this.animator.update(delta);
      }
    }
  }

  public isEnemy(other: CharacterEntity): boolean {
    return (this.faction === 'friendly' && other.faction === 'enemy') ||
           (this.faction === 'enemy' && other.faction === 'friendly');
  }
}
6. 具体角色：友方实体 FriendlyEntity（带动画）

// src/entities/FriendlyEntity.ts
import { CharacterEntity } from './CharacterEntity';
import { PaperAnimator } from '../core/PaperAnimator';
import { AnimationLoader } from '../core/AnimationLoader';
import * as THREE from 'three';

export class FriendlyEntity extends CharacterEntity {
  constructor(id: string, position: THREE.Vector3) {
    // 创建平面几何体和材质
    const geometry = new THREE.PlaneGeometry(1.5, 2); // 纸片人尺寸
    const material = new THREE.MeshStandardMaterial({
      color: 0xffffff,
      side: THREE.DoubleSide,
      transparent: true
    });
    const mesh = new THREE.Mesh(geometry, material);
    super(id, 'friendly', mesh, 'friendly');
    this.position.copy(position);
    this.mesh.position.copy(position);
    this.radius = 0.6;
    this.speed = 5;

    // 创建动画器并注入
    const animator = new PaperAnimator(material);
    this.setAnimator(animator);

    // 异步加载动画（不阻塞初始化）
    this.loadAnimations(animator);
  }

  private async loadAnimations(animator: PaperAnimator): Promise<void> {
    try {
      const { frontClip, backClip } = await AnimationLoader.loadPaperAnimations(
        '/textures/characters/player', // 根据实际路径修改
        8,   // 正面帧数
        8,   // 背面帧数
        12   // 帧率
      );
      animator.setFrontClip(frontClip);
      animator.setBackClip(backClip);
    } catch (error) {
      console.warn('Failed to load animations for player', error);
    }
  }

  // 可选：手动控制移动方向（由外部输入设置）
  public setMoveDirection(x: number, z: number): void {
    this.moveDirection.set(x, 0, z);
  }
}

// src/core/CameraStore.ts
import * as THREE from 'three';

export class CameraStore {
  private static instance: CameraStore;
  private camera: THREE.PerspectiveCamera | null = null;

  private constructor() {}

  public static getInstance(): CameraStore {
    if (!CameraStore.instance) {
      CameraStore.instance = new CameraStore();
    }
    return CameraStore.instance;
  }

  public setCamera(camera: THREE.PerspectiveCamera): void {
    this.camera = camera;
  }

  public getCamera(): THREE.PerspectiveCamera | null {
    return this.camera;
  }
}

export const cameraStore = CameraStore.getInstance();

关键设计总结
动画器可选：只有需要动画的角色才创建 PaperAnimator 并注入，基类 CharacterEntity 提供 animator 字段并在 update 中判断后调用。

异步加载：动画加载不阻塞角色显示，加载完成后自动切换纹理。

方向判断：updateDirection 根据相机与角色的相对位置自动选择正面或背面动画集。

松耦合：动画器不依赖实体具体实现，仅依赖材质和相机位置。

性能：无动画实体零开销；有动画实体仅在活动时更新纹理。

这套代码可以直接复制到你的项目中，只需根据实际图片路径调整 loadPaperAnimations 的参数即可。
