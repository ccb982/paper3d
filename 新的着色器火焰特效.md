用着色器实现火焰特效，可以完全脱离粒子系统，直接在 GPU 中生成动态、细腻的火焰形状和颜色，性能极高，且无需额外纹理。下面给出一个完整的 Three.js 着色器火焰方案。

一、核心思路
几何体：一个简单的平面（PlaneGeometry）或锥体（ConeGeometry），作为火焰的“画布”。

顶点着色器：让几何体顶点随时间波动，模拟火焰摇曳。

片元着色器：根据 UV 坐标和时间，计算颜色和透明度，形成从底部红黄到顶部橙红、边缘半透明的渐变，并叠加噪声制造火焰纹理。

二、完整代码
2.1 火焰着色器材质
typescript
// FireShaderMaterial.ts
import * as THREE from 'three';

export const fireVertexShader = `
  varying vec2 vUv;
  uniform float uTime;
  void main() {
    vUv = uv;
    vec3 pos = position;
    // 根据 UV 的 y 轴（高度）添加顶点波动，模拟火焰摇曳
    float waveX = sin(pos.y * 8.0 + uTime * 10.0) * 0.03;
    float waveZ = cos(pos.y * 7.0 + uTime * 8.0) * 0.03;
    pos.x += waveX;
    pos.z += waveZ;
    vec4 mvPosition = modelViewMatrix * vec4(pos, 1.0);
    gl_PointSize = 1.0;
    gl_Position = projectionMatrix * mvPosition;
  }
`;

export const fireFragmentShader = `
  uniform float uTime;
  varying vec2 vUv;
  
  // 伪随机函数
  float random (vec2 st) {
    return fract(sin(dot(st.xy, vec2(12.9898,78.233))) * 43758.5453123);
  }
  
  // 平滑噪声
  float noise (vec2 st) {
    vec2 i = floor(st);
    vec2 f = fract(st);
    float a = random(i);
    float b = random(i + vec2(1.0, 0.0));
    float c = random(i + vec2(0.0, 1.0));
    float d = random(i + vec2(1.0, 1.0));
    vec2 u = f * f * (3.0 - 2.0 * f);
    return mix(mix(a, b, u.x), mix(c, d, u.x), u.y);
  }
  
  void main() {
    // 基础 UV，让火焰从底部向上燃烧
    float height = vUv.y;  // 0底部 -> 1顶部
    float width = abs(vUv.x - 0.5) * 2.0;  // 0中心 -> 1边缘
    
    // 火焰形状：底部宽，顶部尖
    float shape = 1.0 - smoothstep(0.0, 0.8, width) * (1.0 - height * 0.5);
    shape = clamp(shape, 0.0, 1.0);
    
    // 噪声纹理，制造火焰内部的流动感
    vec2 noiseCoord = vUv * 4.0;
    noiseCoord.x += uTime * 1.5;
    noiseCoord.y -= uTime * 0.8;
    float noiseVal = noise(noiseCoord);
    
    // 额外一层高频噪声
    vec2 noiseCoord2 = vUv * 8.0;
    noiseCoord2.x += uTime * 2.0;
    noiseCoord2.y -= uTime * 1.2;
    float noiseVal2 = noise(noiseCoord2);
    
    float finalNoise = (noiseVal + noiseVal2) * 0.6 + 0.2;
    
    // 根据高度决定颜色（底部红黄，顶部橙红）
    vec3 colorBottom = vec3(1.0, 0.4, 0.1);  // 橙红
    vec3 colorMid = vec3(1.0, 0.8, 0.2);     // 金黄
    vec3 colorTop = vec3(0.8, 0.2, 0.05);    // 暗红
    vec3 color;
    if (height < 0.5) {
      float t = height / 0.5;
      color = mix(colorBottom, colorMid, t);
    } else {
      float t = (height - 0.5) / 0.5;
      color = mix(colorMid, colorTop, t);
    }
    
    // 加入噪声，让颜色有变化
    color += vec3(0.2, 0.1, 0.0) * finalNoise;
    
    // 透明度：底部实，顶部半透明，边缘透明
    float alpha = shape * (0.8 + finalNoise * 0.3);
    alpha *= (1.0 - height * 0.3);  // 顶部略微透明
    
    // 火焰尖端消散效果
    if (height > 0.85) {
      alpha *= (1.0 - (height - 0.85) / 0.15);
    }
    
    gl_FragColor = vec4(color, alpha);
  }
`;

export function createFireMaterial(): THREE.ShaderMaterial {
  return new THREE.ShaderMaterial({
    uniforms: {
      uTime: { value: 0 }
    },
    vertexShader: fireVertexShader,
    fragmentShader: fireFragmentShader,
    transparent: true,
    side: THREE.DoubleSide,
    blending: THREE.AdditiveBlending, // 可选，增强发光感
    depthWrite: false
  });
}
2.2 在场景中使用
typescript
// 创建一个平面或锥体作为火焰几何体
const geometry = new THREE.PlaneGeometry(1, 2, 32, 64);
// 让平面始终面向相机（Billboard）
const material = createFireMaterial();
const flame = new THREE.Mesh(geometry, material);
flame.position.set(0, 0, 0);
scene.add(flame);

// 在动画循环中更新时间
function animate(time: number) {
  material.uniforms.uTime.value = time * 0.001; // 秒
  requestAnimationFrame(animate);
}
三、效果说明
形状：通过 shape 因子控制，底部宽顶部尖，模拟火焰锥形。

颜色：底部橙红，中部金黄，顶部暗红，符合火焰温度分布。

动态：顶点波动 + 噪声纹理流动，产生摇曳感。

透明度：边缘和顶部半透明，中心较实，自然消散。

性能：纯 GPU 计算，无需粒子，性能极好。

你可以根据需要调整参数：

修改 waveX/waveZ 的幅度和频率，改变摇曳强度。

调整噪声坐标的滚动速度，改变火焰内部流动感。

修改颜色和透明度曲线，适配不同火焰风格（魔法火焰、蓝焰等）。
