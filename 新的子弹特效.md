流体模拟画布跟随子弹移动，弹头在画布中的相对位置固定（例如右侧），尾迹向左喷射，看起来像是子弹的喷射尾焰。特效平面会作为子弹的子物体，随子弹移动，流体模拟每帧在画布固定相对位置注入，生成动态尾迹。

一、核心思路
画布不移动：实际上画布就是纹理，附着在平面上，平面移动 → 整个流体效果跟随子弹移动。

弹头相对画布静止：在流体模拟的坐标空间中，把弹头固定在画布右侧（比如 x=400，画宽512），尾迹向左生成（x 递减）。

结果：子弹在世界中移动时，尾迹始终从弹头向后喷出，不会残留在场景中，符合“喷射尾焰”的需求。

二、代码修改
2.1 BulletFluidTexture.ts — 相对坐标注入
关键修改：弹头坐标用常量 BULLET_X 固定，尾迹和涡旋都基于此相对位置生成。

typescript
// BulletFluidTexture.ts 核心部分
import { FluidDynamics } from '@bienehito/fluid-dynamics';
import * as THREE from 'three';

export class BulletFluidTexture {
  private canvas: HTMLCanvasElement;
  private fluidDynamics: FluidDynamics;
  private texture: THREE.CanvasTexture;
  private lastTimestamp: number = 0;
  
  // 固定弹头在画布中的 X 坐标（右侧，留出尾迹空间）
  private readonly BULLET_X = 400;   // 画布宽度 512
  private readonly BULLET_Y = 256;   // 画布中心
  
  constructor() {
    this.canvas = document.createElement('canvas');
    this.canvas.width = 512;
    this.canvas.height = 512;
    
    this.fluidDynamics = new FluidDynamics(this.canvas, {
      width: 512,
      height: 512,
      simScale: 1.0,
      dyeScale: 1.2,
      curl: 12,
      velocityDissipation: 0.05,
      dyeDissipation: 0.08,
      pressureIterations: 20
    });
    
    this.texture = new THREE.CanvasTexture(this.canvas);
    this.animate();
  }
  
  private injectFluid() {
    const x = this.BULLET_X;
    const y = this.BULLET_Y;
    
    // 1. 弹头高压区（白色椭圆）
    this.fluidDynamics.setVelocity(x, y, 0, 25, 15, 0, 0);
    this.fluidDynamics.setDye(x, y, 0, 20, 12, [1, 1, 0.95]);
    
    // 2. 马赫环（弹头前方橙色）
    const aheadX = x + 30;
    const aheadY = y;
    this.fluidDynamics.setVelocity(aheadX, aheadY, 0, 18, 12, 60, 0);
    this.fluidDynamics.setDye(aheadX, aheadY, 0, 15, 10, [1.0, 0.5, 0.2]);
    
    // 3. 锥形尾气核心（向左延伸）
    const trailLength = 200;
    const step = 15;
    for (let dist = 25; dist <= trailLength; dist += step) {
      const backX = x - dist;
      if (backX < 0) continue; // 防止超出左边界
      const width = 8 + dist * 0.15;
      const intensity = 1 - (dist / trailLength);
      
      this.fluidDynamics.setVelocity(backX, y, 0, width, width * 0.6,
        -80 * intensity, (Math.random() - 0.5) * 20);
      this.fluidDynamics.setDye(backX, y, 0, width * 0.8, width * 0.5,
        [0.2, 0.4, 1.0]);
    }
    
    // 4. 卡门涡街（三组交替）
    const vortexOffset = 12;
    const vortexStrength = 150;
    const alternating = Math.sin(Date.now() * 0.015) > 0 ? 1 : -1;
    
    // 第一组涡旋 (距离50)
    const v1X = x - 50;
    if (v1X > 0) {
      this.fluidDynamics.setVelocity(v1X, y + vortexOffset, 0, 15, 15,
        -vortexStrength * alternating, 0);
      this.fluidDynamics.setDye(v1X, y + vortexOffset, 0, 12, 12, [1.0, 0.2, 0.6]);
      this.fluidDynamics.setVelocity(v1X, y - vortexOffset, 0, 15, 15,
        -vortexStrength * alternating, 0);
      this.fluidDynamics.setDye(v1X, y - vortexOffset, 0, 12, 12, [0.1, 0.8, 0.9]);
    }
    
    // 第二组涡旋 (距离90)
    const v2X = x - 90;
    if (v2X > 0) {
      const alternating2 = Math.sin(Date.now() * 0.015 + Math.PI) > 0 ? 1 : -1;
      this.fluidDynamics.setVelocity(v2X, y + vortexOffset * 1.5, 0, 14, 14,
        -vortexStrength * 0.8 * alternating2, 0);
      this.fluidDynamics.setDye(v2X, y + vortexOffset * 1.5, 0, 11, 11, [0.9, 0.3, 1.0]);
      this.fluidDynamics.setVelocity(v2X, y - vortexOffset * 1.5, 0, 14, 14,
        -vortexStrength * 0.8 * alternating2, 0);
      this.fluidDynamics.setDye(v2X, y - vortexOffset * 1.5, 0, 11, 11, [0.1, 0.9, 0.7]);
    }
    
    // 第三组涡旋 (距离130)
    const v3X = x - 130;
    if (v3X > 0) {
      const alternating3 = Math.sin(Date.now() * 0.015 + Math.PI * 0.5) > 0 ? 1 : -1;
      this.fluidDynamics.setVelocity(v3X, y + vortexOffset * 2, 0, 12, 12,
        -vortexStrength * 0.6 * alternating3, 0);
      this.fluidDynamics.setDye(v3X, y + vortexOffset * 2, 0, 10, 10, [0.7, 0.2, 0.9]);
      this.fluidDynamics.setVelocity(v3X, y - vortexOffset * 2, 0, 12, 12,
        -vortexStrength * 0.6 * alternating3, 0);
      this.fluidDynamics.setDye(v3X, y - vortexOffset * 2, 0, 10, 10, [0.2, 0.7, 0.8]);
    }
    
    // 5. 随机湍流
    for (let i = 0; i < 5; i++) {
      const turbX = x - 40 - Math.random() * 150;
      if (turbX < 0) continue;
      const turbY = y + (Math.random() - 0.5) * 60;
      this.fluidDynamics.setVelocity(turbX, turbY, 0, 20, 20,
        (Math.random() - 0.5) * 50, (Math.random() - 0.5) * 50);
    }
  }
  
  private animate() {
    requestAnimationFrame((now) => {
      const delta = Math.min(0.033, (now - this.lastTimestamp) / 1000);
      this.lastTimestamp = now;
      
      // 更新流体模拟（内部消散）
      this.fluidDynamics.update(delta);
      // 注入新流体
      this.injectFluid();
      
      // 更新纹理
      this.texture.needsUpdate = true;
      
      this.animate();
    });
  }
  
  public getTexture(): THREE.CanvasTexture {
    return this.texture;
  }
  
  public dispose() {
    this.fluidDynamics.dispose?.();
    this.texture.dispose();
    this.canvas.remove();
  }
}
2.2 BulletFluidEffect.ts — 特效平面
让平面始终面向相机（billboard），这样尾迹在任何角度都能看到。

typescript
// BulletFluidEffect.ts
import * as THREE from 'three';
import { BulletFluidTexture } from '../systems/textures/BulletFluidTexture';

export class BulletFluidEffect {
  private mesh: THREE.Mesh;
  private textureObj: BulletFluidTexture;
  private material: THREE.MeshBasicMaterial;
  
  constructor(size: number = 2.0) {
    this.textureObj = new BulletFluidTexture();
    const texture = this.textureObj.getTexture();
    
    this.material = new THREE.MeshBasicMaterial({
      map: texture,
      transparent: true,
      blending: THREE.AdditiveBlending,
      opacity: 0.9,
      side: THREE.DoubleSide
    });
    
    const geometry = new THREE.PlaneGeometry(size, size);
    this.mesh = new THREE.Mesh(geometry, this.material);
  }
  
  public getMesh(): THREE.Mesh {
    return this.mesh;
  }
  
  // 让平面始终面向相机（拖尾始终朝向摄像机）
  public lookAtCamera(camera: THREE.Camera) {
    this.mesh.lookAt(camera.position);
  }
  
  public dispose() {
    this.textureObj.dispose();
    this.material.dispose();
    this.mesh.geometry.dispose();
  }
}

2.4 注意事项
平面大小：size 应根据你场景中子弹的大小和期望的尾迹视觉长度调整。如果尾迹显得太短，可以增大 size（比如 3.0 或 4.0），同时调大纹理中 trailLength 到 300 像素。

平面朝向：上述例子使用 lookAtCamera 会使平面一直面向摄像机，这样纹理中的左右方向就会一直平行于屏幕，适合第三人称视角。如果是第一人称，可能需要将平面固定在子弹后方且不旋转。

性能：流体模拟每帧更新画布并上传纹理，开销适中。可以适当降低画布分辨率到 256x256 提升性能。

停止特效：子弹销毁时，调用 fluidEffect.dispose() 释放资源。

三、效果说明
当子弹移动时，流体特效平面跟随，上面动态生成向右喷射的尾迹（纹理中弹头在右侧，尾迹向左）。由于平面面向相机，你会看到完整的彩色涡旋和锥形尾气。

尾迹不会残留在世界中，看起来就像从子弹尾部持续喷出的火焰/流体。

如果你希望尾迹更长，可以调大 trailLength 并适当减小 velocityDissipation。
