# 爆裂黎明特效实现文档

## 一、特效系统架构

### 1. 系统概述
特效系统采用单例模式设计，通过 `EffectManager` 统一管理所有特效的生命周期，支持多种特效类型，包括命中闪光、枪口闪光、爆炸、环形波和爆裂黎明特效。

### 2. 文件结构
```
src/
├── core/
│   └── EffectManager.ts       // 特效管理器
└── effects/
    ├── BaseEffect.ts          // 特效基类
    ├── HitFlashEffect.ts      // 命中闪光
    ├── MuzzleFlashEffect.ts   // 枪口闪光
    ├── ExplosionEffect.ts     // 爆炸
    ├── RingWaveEffect.ts      // 环形波
    └── DawnBurstEffect.ts     // 爆裂黎明特效
```

## 二、核心实现

### 1. 特效基类 (BaseEffect)
```typescript
// src/effects/BaseEffect.ts
export abstract class BaseEffect {
  protected duration: number;
  protected elapsed: number = 0;
  public isActive: boolean = true;

  constructor(duration: number) {
    this.duration = duration;
  }

  public update(delta: number): void {
    if (!this.isActive) return;
    this.elapsed += delta;
    if (this.elapsed >= this.duration) {
      this.isActive = false;
      return;
    }
    this.onUpdate(delta);
  }

  protected abstract onUpdate(delta: number): void;
  public abstract dispose(): void;
}
```

### 2. 特效管理器 (EffectManager)
```typescript
// src/core/EffectManager.ts
import { BaseEffect } from '../effects/BaseEffect';
import { DawnBurstEffect } from '../effects/DawnBurstEffect';

export class EffectManager {
  private static instance: EffectManager;
  private activeEffects: BaseEffect[] = [];
  
  public static getInstance(): EffectManager {
    if (!EffectManager.instance) EffectManager.instance = new EffectManager();
    return EffectManager.instance;
  }
  
  public playDawnExplosion(position: THREE.Vector3): void {
    this.activeEffects.push(new DawnBurstEffect(position, 4.0));
  }
  
  public update(delta: number): void {
    for (let i = this.activeEffects.length-1; i >= 0; i--) {
      const effect = this.activeEffects[i];
      effect.update(delta);
      if (!effect.isActive) {
        effect.dispose();
        this.activeEffects.splice(i, 1);
      }
    }
  }
}
```

### 3. 爆裂黎明特效 (DawnBurstEffect)
```typescript
// src/effects/DawnBurstEffect.ts
import * as THREE from 'three';
import { BaseEffect } from './BaseEffect';
import { EntityManager } from '../core/EntityManager';
import { cameraStore } from '../core/CameraStore';

type Spike = {
  mesh: THREE.Mesh;
  direction: THREE.Vector3;
  speed: number;
  scale: number;
};

type RingEffect = {
  mesh: THREE.Mesh;
  material: THREE.MeshBasicMaterial;
  speed: number;
  maxRadius: number;
  currentRadius: number;
};

export class DawnBurstEffect extends BaseEffect {
  private group: THREE.Group;
  private innerSpikes: Spike[] = [];
  private outerSpikes: Spike[] = [];
  private coreFlash: THREE.Mesh | null = null;
  private coreMaterial: THREE.MeshBasicMaterial | null = null;
  private rings: RingEffect[] = [];
  
  constructor(position: THREE.Vector3, duration: number = 4.0) {
    super(duration);
    this.group = new THREE.Group();
    this.group.position.copy(position);
    const scene = EntityManager.getInstance().getScene();
    if (scene) scene.add(this.group);
    
    // 获取相机方向，使特效朝向相机
    const camera = cameraStore.getCamera();
    let targetDir = new THREE.Vector3(0, 1, 0);
    if (camera) {
      targetDir = new THREE.Vector3().subVectors(camera.position, this.group.position).normalize();
      targetDir.y += 0.2;
      targetDir.normalize();
    }
    
    // 创建核心闪光
    this.createCoreFlash();
    
    // 创建尖刺
    this.createSpikes(targetDir);
    
    // 创建环形波
    this.createRings();
  }
  
  private createCoreFlash(): void {
    const geometry = new THREE.SphereGeometry(0.3, 16, 16);
    this.coreMaterial = new THREE.MeshBasicMaterial({
      color: 0xffffff,
      transparent: true,
      opacity: 1.0,
      blending: THREE.AdditiveBlending
    });
    this.coreFlash = new THREE.Mesh(geometry, this.coreMaterial);
    this.group.add(this.coreFlash);
  }
  
  private createSpikes(direction: THREE.Vector3): void {
    // 内层尖刺（红色）
    for (let i = 0; i < 8; i++) {
      const angle = (i / 8) * Math.PI * 2;
      const spikeDir = new THREE.Vector3(
        Math.cos(angle) * 0.8 + direction.x * 0.2,
        Math.sin(angle) * 0.4 + 0.5,
        Math.sin(angle) * 0.8 + direction.z * 0.2
      ).normalize();
      
      const geometry = new THREE.ConeGeometry(0.05, 0.5, 4);
      const material = new THREE.MeshBasicMaterial({ color: 0xff3366, blending: THREE.AdditiveBlending });
      const mesh = new THREE.Mesh(geometry, material);
      
      // 朝向尖刺方向
      mesh.quaternion.setFromUnitVectors(
        new THREE.Vector3(0, 1, 0),
        spikeDir
      );
      
      this.group.add(mesh);
      this.innerSpikes.push({
        mesh,
        direction: spikeDir,
        speed: 2 + Math.random() * 1,
        scale: 0.5
      });
    }
    
    // 外层尖刺（蓝色）
    for (let i = 0; i < 12; i++) {
      const angle = (i / 12) * Math.PI * 2;
      const spikeDir = new THREE.Vector3(
        Math.cos(angle) + direction.x * 0.1,
        Math.sin(angle) * 0.3 + 0.3,
        Math.sin(angle) + direction.z * 0.1
      ).normalize();
      
      const geometry = new THREE.ConeGeometry(0.03, 0.3, 4);
      const material = new THREE.MeshBasicMaterial({ color: 0x33ccff, blending: THREE.AdditiveBlending });
      const mesh = new THREE.Mesh(geometry, material);
      
      mesh.quaternion.setFromUnitVectors(
        new THREE.Vector3(0, 1, 0),
        spikeDir
      );
      
      this.group.add(mesh);
      this.outerSpikes.push({
        mesh,
        direction: spikeDir,
        speed: 3 + Math.random() * 1.5,
        scale: 0.3
      });
    }
  }
  
  private createRings(): void {
    // 红色环
    this.rings.push(this.createRing(0xff3366, 0.8, 0.1));
    // 蓝色环
    this.rings.push(this.createRing(0x33ccff, 1.0, 0.15));
    // 金色环
    this.rings.push(this.createRing(0xffaa44, 1.2, 0.2));
  }
  
  private createRing(color: number, speed: number, maxRadius: number): RingEffect {
    const geometry = new THREE.RingGeometry(0.01, 0.1, 32);
    const material = new THREE.MeshBasicMaterial({
      color,
      transparent: true,
      opacity: 0.8,
      side: THREE.DoubleSide,
      blending: THREE.AdditiveBlending
    });
    const mesh = new THREE.Mesh(geometry, material);
    mesh.rotation.x = Math.PI / 2; // 水平放置
    this.group.add(mesh);
    
    return {
      mesh,
      material,
      speed,
      maxRadius,
      currentRadius: 0
    };
  }
  
  protected onUpdate(delta: number): void {
    const t = this.elapsed / this.duration;
    
    // 更新核心闪光
    if (this.coreFlash && this.coreMaterial) {
      const scale = 1 + t * 3;
      this.coreFlash.scale.set(scale, scale, scale);
      this.coreMaterial.opacity = 1 - t * 2;
    }
    
    // 更新内层尖刺
    this.innerSpikes.forEach(spike => {
      const distance = spike.speed * delta;
      spike.mesh.position.add(spike.direction.multiplyScalar(distance));
      spike.scale *= 0.98;
      spike.mesh.scale.set(spike.scale, spike.scale, spike.scale);
    });
    
    // 更新外层尖刺
    this.outerSpikes.forEach(spike => {
      const distance = spike.speed * delta;
      spike.mesh.position.add(spike.direction.multiplyScalar(distance));
      spike.scale *= 0.97;
      spike.mesh.scale.set(spike.scale, spike.scale, spike.scale);
    });
    
    // 更新环形波
    this.rings.forEach(ring => {
      ring.currentRadius += ring.speed * delta;
      if (ring.currentRadius > ring.maxRadius) {
        ring.currentRadius = ring.maxRadius;
      }
      
      const scale = ring.currentRadius / 0.1;
      ring.mesh.scale.set(scale, scale, 1);
      ring.material.opacity = 0.8 * (1 - ring.currentRadius / ring.maxRadius);
    });
  }
  
  public dispose(): void {
    // 清理核心闪光
    if (this.coreFlash) {
      this.group.remove(this.coreFlash);
      this.coreFlash.geometry.dispose();
      if (this.coreMaterial) {
        this.coreMaterial.dispose();
      }
    }
    
    // 清理尖刺
    this.innerSpikes.forEach(spike => {
      this.group.remove(spike.mesh);
      spike.mesh.geometry.dispose();
      (spike.mesh.material as THREE.Material).dispose();
    });
    
    this.outerSpikes.forEach(spike => {
      this.group.remove(spike.mesh);
      spike.mesh.geometry.dispose();
      (spike.mesh.material as THREE.Material).dispose();
    });
    
    // 清理环形波
    this.rings.forEach(ring => {
      this.group.remove(ring.mesh);
      ring.mesh.geometry.dispose();
      ring.material.dispose();
    });
    
    // 从场景中移除整个组
    const scene = EntityManager.getInstance().getScene();
    if (scene) {
      scene.remove(this.group);
    }
  }
}
```

### 4. 爆裂黎明子弹实体
```typescript
// src/entities/bullets/DawnExplosionBulletEntity.ts
import * as THREE from 'three';
import { BulletEntity } from './BulletEntity';
import { EffectManager } from '../../core/EffectManager';
import { EntityManager } from '../../core/EntityManager';
import { CharacterEntity } from '../characters/CharacterEntity';
import { StaticEntity } from '../static/StaticEntity';

export class DawnExplosionBulletEntity extends BulletEntity {
  private explosionRadius: number = 10; // 爆炸范围半径
  
  constructor(
    position: THREE.Vector3,
    direction: THREE.Vector3,
    speed: number,
    color: number = 0xffd700
  ) {
    super(position, direction, speed, color);
    this.setDamage(2);
  }
  
  protected onHit(entity: any): void {
    super.onHit(entity);
    EffectManager.getInstance().playDawnExplosion(this.position);
    this.applyAreaDamage();
  }
  
  private applyAreaDamage(): void {
    const allEntities = EntityManager.getInstance().getAllEntities();
    const areaDamage = this.getDamage() * 0.5;
    
    allEntities.forEach((entity) => {
      if (entity.id === this.id || !entity.isActive) return;
      
      const distance = entity.position.distanceTo(this.position);
      if (distance <= this.explosionRadius) {
        if (
          (entity instanceof CharacterEntity && (entity as CharacterEntity).faction === 'enemy') ||
          (entity instanceof StaticEntity && (entity as StaticEntity).isShootable)
        ) {
          const damageMultiplier = 1 - (distance / this.explosionRadius);
          const finalDamage = Math.round(areaDamage * damageMultiplier);
          
          if (finalDamage > 0 && typeof entity.takeDamage === 'function') {
            entity.takeDamage(finalDamage);
          }
        }
      }
    });
  }
}
```

## 三、集成使用

### 1. 在游戏世界中注册碰撞回调
```typescript
// src/components/GameWorld.tsx
// 注册子弹 vs 角色（敌人）碰撞
collisionManager.registerCollision('bullet', 'character', (bullet, character) => {
  const bulletEntity = bullet as BulletEntity;
  const characterEntity = character as CharacterEntity;
  if (characterEntity.faction === 'enemy') {
    const currentMode = useGameStore.getState().mode;
    if (currentMode !== GameMode.BATTLE) {
      useGameStore.getState().setMode(GameMode.BATTLE);
    }
    // 调用子弹的onHit方法，触发范围伤害
    if (typeof (bulletEntity as any).onHit === 'function') {
      (bulletEntity as any).onHit(characterEntity);
    }
    bulletEntity.isActive = false;
  }
});
```

## 四、效果描述

### 1. 视觉效果
- **核心爆闪**：中心白色闪光球体，快速膨胀并淡出
- **多层尖刺**：内层红色尖刺和外层蓝色尖刺，向四周喷射
- **多环冲击波**：红、蓝、金三层圆环，从中心向外扩散
- **范围伤害**：爆炸范围内的敌人和静态实体受到伤害

### 2. 技术特点
- **面向相机**：特效自动朝向相机方向，确保最佳视觉效果
- **范围伤害**：实现10单位半径的范围伤害，对爆炸范围内的敌人和静态实体造成伤害
- **性能优化**：自动管理特效生命周期，及时清理不再活跃的特效
- **纯程序化**：无需外部贴图，完全通过Three.js几何体和材质实现

## 五、自定义选项

1. **爆炸范围**：修改 `DawnExplosionBulletEntity` 中的 `explosionRadius` 属性
2. **特效持续时间**：修改 `DawnBurstEffect` 构造函数中的 `duration` 参数
3. **颜色方案**：修改 `DawnBurstEffect` 中尖刺和环形波的颜色
4. **尖刺数量**：调整 `createSpikes` 方法中的循环次数
5. **伤害系数**：修改 `applyAreaDamage` 方法中的伤害计算逻辑

## 六、性能考虑

- **特效生命周期管理**：EffectManager 自动清理过期特效，避免内存泄漏
- **几何体复用**：对于简单几何体（如尖刺、环形波），可以考虑使用几何体池
- **材质共享**：相同颜色和属性的材质可以共享，减少内存使用
- **特效数量控制**：避免在短时间内触发过多特效，可能导致性能下降
