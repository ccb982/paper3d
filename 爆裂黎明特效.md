想要的是一个类似银河系旋涡星云的精细特效，而不是简单的圆环。这需要多层粒子系统 + 螺旋密度分布 + 颜色渐变 + 光晕效果。下面给出基于你现有 DawnBurstEffect 框架的完整实现。

设计思路
粒子数量：数千至上万个粒子，构成旋臂结构。

螺旋模型：使用对数螺旋（r = a * e^(b * theta)）让粒子从中心向外辐射，且旋臂越往外越稀疏。

多层旋臂：至少 2-4 条主旋臂，每条旋臂可分支或添加随机扰动。

密度分布：内圈粒子密集，外圈稀疏；沿旋臂方向粒子密度高，臂间空隙有少量弥散粒子。

颜色：从中心暖黄色/橙色向外渐变为蓝色/紫色，模拟银河系。

动态效果：整体缓慢旋转，粒子有微小脉动（呼吸感）。

光晕：使用 AdditiveBlending 叠加，粒子带发光感。

代码实现
以下代码完全独立于你之前的尖刺和环，可以直接添加到 DawnBurstEffect 中作为额外特效层。你也可以单独创建一个 GalaxyBurstEffect 类。

1. 银河粒子生成器
typescript
// 生成银河旋涡粒子数据
function generateGalaxyParticles(
  particleCount: number,
  arms: number = 3,            // 旋臂数量
  innerRadius: number = 0.5,
  outerRadius: number = 8,
  armSpread: number = 0.4,     // 旋臂的宽度（角度偏移范围）
  armTightness: number = 3,    // 螺旋紧密程度（对数螺旋的系数）
  heightRange: number = 1.2    // 垂直方向范围
): { positions: Float32Array; colors: Float32Array; sizes: Float32Array } {
  const positions = new Float32Array(particleCount * 3);
  const colors = new Float32Array(particleCount * 3);
  const sizes = new Float32Array(particleCount);

  // 颜色映射函数：半径比例 -> 颜色 (HSL)
  const getColor = (r: number) => {
    // r 范围 innerRadius -> outerRadius，映射到 0..1
    const t = (r - innerRadius) / (outerRadius - innerRadius);
    // 色相：从 0.12 (橙黄) 到 0.6 (蓝紫)
    const hue = 0.12 + t * 0.48;
    // 饱和度：内圈高，外圈稍低
    const saturation = 0.7 - t * 0.2;
    // 亮度：内圈亮，外圈稍暗但仍有光晕
    const lightness = 0.6 + (1 - t) * 0.3;
    return new THREE.Color().setHSL(hue, saturation, lightness);
  };

  for (let i = 0; i < particleCount; i++) {
    // 决定粒子属于哪条旋臂（均匀分配）
    const armIdx = Math.floor(Math.random() * arms);
    const armAngleOffset = (armIdx / arms) * Math.PI * 2;

    // 半径：使用平方分布，让更多粒子在外围（模拟旋臂延伸）
    let r = innerRadius + (outerRadius - innerRadius) * Math.pow(Math.random(), 1.5);
    // 对数螺旋角度：theta = k * ln(r / a)
    // 简化：theta = armTightness * (r - innerRadius) / (outerRadius - innerRadius) * 2pi
    const baseAngle = armTightness * (r - innerRadius) / (outerRadius - innerRadius) * Math.PI * 2;
    // 添加旋臂宽度随机偏移（使旋臂有厚度）
    const armOffset = (Math.random() - 0.5) * armSpread * (1 - (r - innerRadius)/(outerRadius - innerRadius)) * 1.5;
    // 随机扰动，增加自然感
    const randomJitter = (Math.random() - 0.5) * 0.3;
    const angle = baseAngle + armAngleOffset + armOffset + randomJitter;

    // 计算坐标
    const x = Math.cos(angle) * r;
    const y = Math.sin(angle) * r;
    // 高度随半径增加而略微增加，但整体扁平，再添加随机偏移
    const z = (Math.random() - 0.5) * heightRange * (1 - (r - innerRadius)/(outerRadius - innerRadius) * 0.5);

    positions[i*3] = x;
    positions[i*3+1] = y;
    positions[i*3+2] = z;

    // 颜色基于半径和旋臂位置微调
    const baseColor = getColor(r);
    // 根据旋臂偏移微调色相
    let hueShift = (armOffset * 0.5);
    const finalColor = new THREE.Color().setHSL(
      Math.min(0.8, Math.max(0.0, baseColor.getHSL({}).h + hueShift)),
      baseColor.getHSL({}).s,
      baseColor.getHSL({}).l
    );
    colors[i*3] = finalColor.r;
    colors[i*3+1] = finalColor.g;
    colors[i*3+2] = finalColor.b;

    // 大小：内圈粒子稍大，外圈稍小，旋臂上粒子略大
    let size = 0.08 + Math.random() * 0.1;
    if (Math.abs(armOffset) < 0.2) size += 0.03;
    sizes[i] = size;
  }

  return { positions, colors, sizes };
}
2. 着色器材质（带光晕和脉冲）
glsl
// 顶点着色器 (galaxyVertexShader)
uniform float uTime;
attribute float size;
attribute vec3 color;
varying vec3 vColor;

void main() {
    vColor = color;
    vec3 pos = position;
    // 微小脉动：让粒子沿径向轻微运动
    float pulse = sin(uTime * 2.0 + length(pos) * 5.0) * 0.02;
    pos += normalize(pos) * pulse;
    
    vec4 mvPosition = modelViewMatrix * vec4(pos, 1.0);
    gl_PointSize = size * ( 300.0 / ( - mvPosition.z ) );
    gl_Position = projectionMatrix * mvPosition;
}
glsl
// 片元着色器 (galaxyFragmentShader)
uniform float uTime;
varying vec3 vColor;

void main() {
    vec2 coord = gl_PointCoord;
    float dist = length(coord - 0.5);
    if (dist > 0.5) discard;
    // 高斯光晕效果
    float alpha = (1.0 - smoothstep(0.0, 0.5, dist)) * 0.9;
    // 中心亮白，边缘带颜色
    vec3 finalColor = vColor + vec3(0.6, 0.4, 0.2) * (1.0 - dist * 1.5);
    gl_FragColor = vec4(finalColor, alpha);
}
3. 集成到 DawnBurstEffect
在 DawnBurstEffect 类中添加成员 galaxyPoints: THREE.Points | null，在构造函数中调用 createGalaxy()，并在 onUpdate 中更新时间 uniform。

typescript
// 在 DawnBurstEffect 类中添加
private galaxyPoints: THREE.Points | null = null;

private createGalaxy(): void {
  const particleCount = 15000;  // 粒子数量，可调整
  const { positions, colors, sizes } = generateGalaxyParticles(
    particleCount, 4,          // 4条旋臂
    0.8, 7.5, 0.45, 3.2, 1.0
  );
  const geometry = new THREE.BufferGeometry();
  geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
  geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
  geometry.setAttribute('size', new THREE.BufferAttribute(sizes, 1));

  const material = new THREE.ShaderMaterial({
    uniforms: { uTime: { value: 0 } },
    vertexShader: galaxyVertexShader,
    fragmentShader: galaxyFragmentShader,
    transparent: true,
    blending: THREE.AdditiveBlending,
    depthWrite: false,
  });

  this.galaxyPoints = new THREE.Points(geometry, material);
  this.group.add(this.galaxyPoints);
}

// 在构造函数中调用 this.createGalaxy();
// 在 onUpdate 中更新时间：
if (this.galaxyPoints) {
  (this.galaxyPoints.material as THREE.ShaderMaterial).uniforms.uTime.value = this.elapsed;
}

private createGlow(): void {
  const count = 3000;
  const positions = new Float32Array(count * 3);
  const colors = new Float32Array(count * 3);
  for (let i = 0; i < count; i++) {
    const r = 5 + Math.random() * 4;
    const angle = Math.random() * Math.PI * 2;
    const x = Math.cos(angle) * r;
    const y = Math.sin(angle) * r;
    const z = (Math.random() - 0.5) * 2.5;
    positions[i*3] = x;
    positions[i*3+1] = y;
    positions[i*3+2] = z;
    const color = new THREE.Color().setHSL(0.6, 0.6, 0.3);
    colors[i*3] = color.r;
    colors[i*3+1] = color.g;
    colors[i*3+2] = color.b;
  }
  const geometry = new THREE.BufferGeometry();
  geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
  geometry.setAttribute('color', new THREE.BufferAttribute(colors, 3));
  const material = new THREE.PointsMaterial({ color: 0x88aaff, size: 0.08, transparent: true, opacity: 0.3, blending: THREE.AdditiveBlending });
  const points = new THREE.Points(geometry, material);
  this.group.add(points);
}
效果预览
旋臂结构：4条对数螺旋旋臂，粒子沿臂密集分布，臂间有少量弥散粒子。

色彩渐变：从中心金黄橙色渐变到外围蓝紫色，饱和度亮度动态变化。

立体感：粒子在Z轴上有轻微高度分布，内圈扁平，外圈稍厚。

动态：整体缓慢旋转（你可以给 this.group.rotation.y 加一个随时间增加的偏移），粒子有径向脉动。

性能优化建议
如果粒子数量过多导致掉帧，可减少到 8000 或 6000，仍能保持旋臂感。

使用 BufferGeometry 一次性上传数据，避免每帧更新位置。

着色器中的 uTime 只控制脉动，开销极小。

现在你的爆炸特效将拥有银河般的华丽旋涡，远超简单圆环。你可以进一步调整参数（旋臂数量、紧密程度、颜色范围）来匹配你心目中的“爆裂黎明”。
