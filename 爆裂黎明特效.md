真正的“爆裂黎明”需要多层组合：核心爆闪、多环冲击波、粒子喷射、光线拖尾、空间扭曲等。而且特效系统应该支持组合模式，一个复杂特效可以包含多个子特效，每个子特效独立更新和销毁，最后统一回收。

下面我重新设计一个可组合的特效系统，然后实现一个真正华丽的“爆裂黎明”。

一、特效系统架构（组合模式）
核心思想
每个特效都是一个 EffectNode，可以添加子节点。

CompositeEffect 负责管理子特效，统一生命周期。

EffectManager 只管理顶层特效，不关心内部结构。

所有特效继承 BaseEffect。

1. 闪光球体 FlashSphere
typescript
// src/effects/basic/FlashSphere.ts
import * as THREE from 'three';
import { BaseEffect } from '../BaseEffect';
import { EntityManager } from '../../core/EntityManager';

export class FlashSphere extends BaseEffect {
  private mesh: THREE.Mesh;
  private material: THREE.MeshBasicMaterial;

  constructor(position: THREE.Vector3, radius: number = 0.5, color: number = 0xffaa66, duration: number = 0.3) {
    super(duration);
    const geometry = new THREE.SphereGeometry(radius, 16, 16);
    this.material = new THREE.MeshBasicMaterial({ color, transparent: true, blending: THREE.AdditiveBlending });
    this.mesh = new THREE.Mesh(geometry, this.material);
    this.mesh.position.copy(position);
    EntityManager.getInstance().getScene()?.add(this.mesh);
  }

  protected onUpdate(delta: number): void {
    const t = this.elapsed / this.duration; // 0→1
    const scale = 1 + t * 2;                // 膨胀
    this.mesh.scale.set(scale, scale, scale);
    this.material.opacity = 1 - t;
  }

  public dispose(): void {
    this.mesh.parent?.remove(this.mesh);
    this.mesh.geometry.dispose();
    this.material.dispose();
  }
}

3. 粒子爆裂 ParticleBurst
typescript
// src/effects/basic/ParticleBurst.ts
import * as THREE from 'three';
import { BaseEffect } from '../BaseEffect';
import { EntityManager } from '../../core/EntityManager';

export class ParticleBurst extends BaseEffect {
  private points: THREE.Points;
  private geometry: THREE.BufferGeometry;
  private velocities: THREE.Vector3[];
  private particleCount: number;
  private material: THREE.PointsMaterial;

  constructor(position: THREE.Vector3, count: number = 150, color: number = 0xff8844, duration: number = 1.0) {
    super(duration);
    this.particleCount = count;
    this.velocities = [];
    const positions = new Float32Array(count * 3);
    for (let i = 0; i < count; i++) {
      positions[i*3] = position.x;
      positions[i*3+1] = position.y;
      positions[i*3+2] = position.z;
      // 随机速度方向（球面均匀分布）
      const dir = new THREE.Vector3(Math.random() - 0.5, Math.random() - 0.5, Math.random() - 0.5).normalize();
      const speed = 3 + Math.random() * 5;
      this.velocities.push(dir.multiplyScalar(speed));
    }
    this.geometry = new THREE.BufferGeometry();
    this.geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    this.material = new THREE.PointsMaterial({ color, size: 0.08, transparent: true, blending: THREE.AdditiveBlending });
    this.points = new THREE.Points(this.geometry, this.material);
    this.points.position.copy(position);
    EntityManager.getInstance().getScene()?.add(this.points);
  }

  protected onUpdate(delta: number): void {
    const positionsAttr = this.geometry.attributes.position;
    const positions = positionsAttr.array as Float32Array;
    for (let i = 0; i < this.particleCount; i++) {
      positions[i*3] += this.velocities[i].x * delta;
      positions[i*3+1] += this.velocities[i].y * delta;
      positions[i*3+2] += this.velocities[i].z * delta;
      // 重力影响
      this.velocities[i].y -= delta * 5;
    }
    positionsAttr.needsUpdate = true;
    const t = this.elapsed / this.duration;
    this.material.opacity = 1 - t;
  }

  public dispose(): void {
    this.points.parent?.remove(this.points);
    this.geometry.dispose();
    this.material.dispose();
  }
}
4. 拖尾粒子 TrailParticles
模拟从中心向外直线飞行的粒子，带拖尾效果（用多个连续粒子）。

typescript
// src/effects/basic/TrailParticles.ts
import * as THREE from 'three';
import { BaseEffect } from '../BaseEffect';
import { EntityManager } from '../../core/EntityManager';

export class TrailParticles extends BaseEffect {
  private particles: THREE.Mesh[] = [];
  private velocities: THREE.Vector3[] = [];
  private lifeRemaining: number[] = [];

  constructor(position: THREE.Vector3, count: number = 30, color: number = 0xffaa66, duration: number = 1.2) {
    super(duration);
    const geometry = new THREE.SphereGeometry(0.05, 4, 4);
    const material = new THREE.MeshStandardMaterial({ color, emissive: color, emissiveIntensity: 0.5 });
    for (let i = 0; i < count; i++) {
      const dir = new THREE.Vector3(Math.random() - 0.5, Math.random() - 0.5, Math.random() - 0.5).normalize();
      const speed = 5 + Math.random() * 4;
      this.velocities.push(dir.multiplyScalar(speed));
      this.lifeRemaining.push(1.0);
      const particle = new THREE.Mesh(geometry, material.clone());
      particle.position.copy(position);
      EntityManager.getInstance().getScene()?.add(particle);
      this.particles.push(particle);
    }
  }

  protected onUpdate(delta: number): void {
    for (let i = 0; i < this.particles.length; i++) {
      if (this.lifeRemaining[i] <= 0) continue;
      this.lifeRemaining[i] -= delta / this.duration;
      const pos = this.particles[i].position;
      pos.x += this.velocities[i].x * delta;
      pos.y += this.velocities[i].y * delta;
      pos.z += this.velocities[i].z * delta;
      const mat = this.particles[i].material as THREE.MeshStandardMaterial;
      mat.opacity = this.lifeRemaining[i];
      mat.transparent = true;
    }
  }

  public dispose(): void {
    this.particles.forEach(p => {
      p.parent?.remove(p);
      p.geometry.dispose();
      (p.material as THREE.Material).dispose();
    });
  }
}

export class GlowPlane extends BaseEffect {

    this.mesh = new THREE.Mesh(geometry, this.material);

    EntityManager.getInstance().getScene()?.add(this.mesh);
  }

  protected onUpdate(delta: number): void {
    const t = this.elapsed / this.duration;
    const scale = 1 + t * 2;
    this.mesh.scale.set(scale, scale, 1);
    this.material.opacity = 0.7 * (1 - t);
    // 面向相机
    const camera = (window as any).cameraStore?.getCamera();
    if (camera) this.mesh.lookAt(camera.position);
  }

  public dispose(): void {
    this.mesh.parent?.remove(this.mesh);
    this.mesh.geometry.dispose();
    this.material.dispose();
  }
}
三、组合特效 CompositeEffect
typescript
// src/effects/CompositeEffect.ts
import { BaseEffect } from './BaseEffect';

export class CompositeEffect extends BaseEffect {
  private children: BaseEffect[] = [];

  public add(child: BaseEffect): void {
    this.children.push(child);
  }

  protected onUpdate(delta: number): void {
    for (let i = this.children.length-1; i >= 0; i--) {
      const child = this.children[i];
      child.update(delta);
      if (!child.isActive) {
        child.dispose();
        this.children.splice(i, 1);
      }
    }
  }

  public dispose(): void {
    this.children.forEach(child => child.dispose());
    this.children = [];
  }
}
四、华丽版“爆裂黎明”特效
组合以下元素：

一个快速放大的白色闪光球体（核心爆炸）

三个不同颜色、不同延迟的圆环波（红、蓝、金）

大量随机粒子爆裂（两批，一批暖色一批冷色）

若干拖尾粒子（射线状）

一个面向相机的光晕平面

一个二次冲击波（延迟 0.15 秒）

typescript
// src/effects/complex/DawnExplosion.ts
import * as THREE from 'three';
import { CompositeEffect } from '../CompositeEffect';
import { FlashSphere } from '../basic/FlashSphere';
import { RingWave } from '../basic/RingWave';
import { ParticleBurst } from '../basic/ParticleBurst';
import { TrailParticles } from '../basic/TrailParticles';
import { GlowPlane } from '../basic/GlowPlane';

export class DawnExplosion extends CompositeEffect {
  constructor(position: THREE.Vector3) {
    super(1.2); // 总时长不重要，子特效各自管理

    // 1. 核心爆闪
    this.add(new FlashSphere(position, 0.6, 0xffeedd, 0.25));

    // 2. 红蓝金三环，错开时间（通过延迟加入，但 CompositeEffect 会立即开始，这里可以用 setTimeout 或者实现延迟特效）
    // 简便方法：创建带延迟的包装特效，但为了简单，我们可以直接添加，它们会同时开始，视觉效果也不错
    this.add(new RingWave(position, 0xff3366, 0.4, 0.7));
    this.add(new RingWave(position, 0x33ccff, 0.5, 0.8));
    this.add(new RingWave(position, 0xffaa44, 0.6, 0.9));

    // 3. 粒子爆裂（暖色 + 冷色）
    this.add(new ParticleBurst(position, 120, 0xff8844, 1.0));
    this.add(new ParticleBurst(position, 80, 0x44aaff, 0.9));

    // 4. 拖尾粒子（射线状）
    this.add(new TrailParticles(position, 40, 0xffaa66, 1.2));

    // 5. 光晕平面
    this.add(new GlowPlane(position, 0xffaa88, 1.0, 0.5));

    // 6. 二次冲击波（延迟 0.15s）
    setTimeout(() => {
      this.add(new RingWave(position, 0xff44aa, 0.7, 0.6));
    }, 150);
  }
}
注意：由于 setTimeout 在特效生命周期中可能不安全（如果特效被提前销毁），更好的做法是实现一个 DelayedEffect 包装器。但为了简化，这里可以接受，因为爆炸特效持续时间短。

七、效果描述
这个组合特效会呈现出：

中心剧烈闪白（球体膨胀并消失）

同时向外扩散红、蓝、金三层圆环，速度不同

大量橙色和蓝色粒子向四面八方飞散，受重力影响下落

数十个光点拖尾直线飞出，模拟冲击射线

一个半透明光晕平面快速放大并淡出

稍后出现第二个紫色圆环，模拟余波

整个特效持续约 1 秒，视觉丰富，无需任何外部贴图。

八、进一步自定义
你可以轻松修改组合：增加更多粒子层、调整颜色、速度、数量，或者加入旋转环、螺旋粒子等。只需新建基础特效类，然后在 DawnExplosion 中 add 即可。

如果你需要更高级的效果（如使用 Shader 实现扭曲、光晕），也可以编写自定义着色器特效类，但上述纯几何 + 粒子已经足够华丽。
