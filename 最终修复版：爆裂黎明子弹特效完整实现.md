  const uvs = [];
  const layerIndices = [];
  const layerOffset = 0;
  const scaleFactor = 0.7;
  
  // 三个形状的点，顺序完全正确
  const mainPoints: [number, number][] = [
    [0, 0], [-0.1, -0.1], [-0.15, -0.2], [-0.2, -0.5], [0.2, -0.5], [0.15, -0.2], [0.1, -0.1]

export function createBulletTrailMaterial(texture: THREE.Texture): THREE.ShaderMaterial {
  texture.wrapS = THREE.ClampToEdgeWrapping;
  texture.wrapT = THREE.ClampToEdgeWrapping;
  
  const vertexShader = `
    uniform float uTime;
    uniform float uWobbleIntensity;
    uniform float uWobbleSpeed;
    uniform float uBulletSpeed;
    varying vec2 vUv;
    varying float vPositionZ;
    attribute float layer;
    
    void main() {
      vUv = uv;
      vPositionZ = position.z;
      vec3 pos = position;
      
      float normalizedZ = position.z - 2.0;
      float tailFactor = 1.0 - normalizedZ;
      // 速度联动：速度越快，摆动越强
      float speedFactor = 1.0 + 0.5 * uBulletSpeed / 500.0;
      float wobbleAmplitude = pow(tailFactor, 3.0) * uWobbleIntensity * 1.5 * speedFactor;
      float wobbleFreq = uWobbleSpeed * speedFactor;
      
      if (layer < 4.0) {
        // 主尾气的4层，湍流摆动
        float layerFactor = 1.0 - layer * 0.2;
        float wobbleX = sin(uTime * wobbleFreq * 1.5 * layerFactor + tailFactor * 12.0) * wobbleAmplitude * 0.8;
        wobbleX += sin(uTime * wobbleFreq * 2.0 * layerFactor + tailFactor * 18.0) * wobbleAmplitude * 0.4;
        float wobbleY = cos(uTime * wobbleFreq * 1.2 * layerFactor + tailFactor * 10.0) * wobbleAmplitude * 0.6;
        wobbleY += cos(uTime * wobbleFreq * 1.6 * layerFactor + tailFactor * 14.0) * wobbleAmplitude * 0.3;
        pos.x += wobbleX;
        pos.y += wobbleY;
      } else {
        // 拖尾的3层，卡门涡街交替摆动
        float layerFactor = 1.0 - (layer - 4.0) * 0.2;
        // 左右拖尾的相位差，模拟交替涡旋
        float phase = vUv.x < 0.5 ? 0.785 : -0.785;
        float wobbleX = sin(uTime * wobbleFreq * 0.8 * layerFactor + phase + tailFactor * 8.0) * wobbleAmplitude * 1.2;
        wobbleX += sin(uTime * wobbleFreq * 1.0 * layerFactor + phase + tailFactor * 12.0) * wobbleAmplitude * 0.6;
        float wobbleY = cos(uTime * wobbleFreq * 0.6 * layerFactor + phase + tailFactor * 6.0) * wobbleAmplitude * 1.0;
        wobbleY += cos(uTime * wobbleFreq * 0.8 * layerFactor + phase + tailFactor * 10.0) * wobbleAmplitude * 0.5;
        pos.x += wobbleX;
        pos.y += wobbleY;
      }
      
      gl_Position = projectionMatrix * modelViewMatrix * vec4(pos, 1.0);
    }
  `;
  
  const fragmentShader = `
    uniform sampler2D uTexture;
    uniform float uTime;
    uniform float uDistortStrength;
    varying vec2 vUv;
    varying float vPositionZ;
    
    void main() {
      // 轻微的热扭曲，模拟高温气体折射
      vec2 uv = vUv;
      vec2 distort = uv - 0.5;
      float dist = length(distort);
      if (dist < 0.5) {
        float strength = (1.0 - dist * 2.0) * uDistortStrength;
        uv += distort * strength;
      }
      
      vec4 texColor = texture2D(uTexture, uv);
      if (texColor.a < 0.1) {
        discard;
      }
      
      // 前端发光，速度联动
      float normalizedZ = vPositionZ - 2.0;
      if (normalizedZ > 0.95 && abs(vUv.x - 0.5) < 0.1) {
        float glowIntensity = (normalizedZ - 0.95) * 20.0;
        float widthIntensity = 1.0 - abs(vUv.x - 0.5) / 0.1;
        vec3 glowColor = vec3(0.8, 0.6, 0.8);
        texColor.rgb = mix(texColor.rgb, glowColor, glowIntensity * widthIntensity * 0.5);
        texColor.a = mix(texColor.a, 1.0, glowIntensity * widthIntensity * 0.3);
      }
      
      // Bloom预乘，让亮的地方自动辉光
      texColor.rgb *= 1.2;
      
      gl_FragColor = texColor;
    }
  `;
  
  return new THREE.ShaderMaterial({
    uniforms: {
      uTexture: { value: texture },
      uTime: { value: 0 },
      uWobbleIntensity: { value: 0.1 },
      uWobbleSpeed: { value: 4.0 },
      uBulletSpeed: { value: 300.0 },
      uDistortStrength: { value: 0.01 }
    },
    vertexShader,
    fragmentShader,
    transparent: true,
    side: THREE.DoubleSide,
    blending: THREE.AdditiveBlending
  });
}

export class DawnExplosionBulletEntity {
  private trailTime: number = 0;
  private trailMesh: THREE.Mesh | null = null;
  private trailMaterial: THREE.ShaderMaterial | null = null;
  private cameraDistance: number = 0;
  
  constructor(
    position: THREE.Vector3,
    direction: THREE.Vector3,
    speed: number,
    color: number = 0xdd5b42
  ) {
    super(position, direction, speed, color);
    this.setDamage(2);
    this.createTrailEffect();
    this.modifyBulletMaterial(speed);
  }
  
  private createTrailEffect(): void {
    const textureManager = new TextureManager();
    createBulletTrailTexture(textureManager);
    const texture = textureManager.getTexture('bullet-trail');
    if (!texture) return;
    
    const geometry = createBulletTrailGeometry();
    this.trailMaterial = createBulletTrailMaterial(texture);
    this.trailMesh = new THREE.Mesh(geometry, this.trailMaterial);
    this.trailMesh.scale.set(3, 3, 3);
    
    const scene = EntityManager.getInstance().getScene();
    if (scene) {
      scene.add(this.trailMesh);
    }
    
    const flightDir = this.velocity.clone().normalize();
    const distance = 8.5;
    this.trailMesh.position.copy(this.position.clone().add(flightDir.clone().multiplyScalar(-distance)));
    this.trailMesh.quaternion.copy(this.mesh.quaternion);
  }
  
  private modifyBulletMaterial(speed: number): void {
    const textureManager = new TextureManager();
    createBulletTrailTexture(textureManager);
    const trailTexture = textureManager.getTexture('bullet-trail');
    if (!trailTexture) return;
    
    // 弹头的激波材质，带热扭曲和速度发光
    const bulletMaterial = new THREE.ShaderMaterial({
      uniforms: {
        baseColor: { value: new THREE.Color(this.mesh.material instanceof THREE.MeshStandardMaterial ? this.mesh.material.color : 0xdd5b42) },
        trailTexture: { value: trailTexture },
        time: { value: 0 },
        speed: { value: speed },
        distortStrength: { value: 0.015 }
      },
      vertexShader: `
        varying vec2 vUv;
        varying vec3 vPosition;
        void main() {
          vUv = uv;
          vPosition = position;
          gl_Position = projectionMatrix * modelViewMatrix * vec4(position, 1.0);
        }
      `,
      fragmentShader: `
        uniform vec3 baseColor;
        uniform sampler2D trailTexture;
        uniform float time;
        uniform float speed;
        uniform float distortStrength;
        varying vec2 vUv;
        varying vec3 vPosition;
        
        void main() {
          vec3 color = baseColor;
          
          // 弹头的热扭曲
          vec2 uv = vUv;
          vec2 distort = uv - 0.5;
          float dist = length(distort);
          if (dist < 0.5) {
            float strength = (1.0 - dist * 2.0) * distortStrength * (1.0 + 0.2 * speed / 500.0);
            uv += distort * strength;
          }
          
          // 前端尾气流动
          float trailLength = 0.3;
          float zNormalized = (vPosition.z + 0.3) / 0.6;
          float textureY = zNormalized * trailLength;
          float timeOffset = mod(time * speed * 0.1, trailLength);
          float trailUvY = textureY + timeOffset;
          trailUvY = mod(trailUvY, 1.0);
          vec2 trailUv = vec2(uv.x, trailUvY);
          vec4 trailColor = texture2D(trailTexture, trailUv);
          
          float alpha = trailColor.a * 0.8;
          color = mix(color, trailColor.rgb, alpha);
          
          // 速度联动的发光
          if (zNormalized > 0.95) {
            float glowIntensity = (zNormalized - 0.95) * 20.0 * (1.0 + 0.2 * speed / 500.0);
            vec3 glowColor = vec3(0.8, 0.6, 0.8);
            color = mix(color, glowColor, glowIntensity * 0.5);
          }
          
          // 激波的红色过渡
          if (zNormalized < 0.3) {
            float endIntensity = 1.0 - (zNormalized / 0.3);
            vec3 endColor = vec3(1.0, 0.0, 0.21);
            color = mix(color, endColor, endIntensity * 0.8);
          }
          
          gl_FragColor = vec4(color, 1.0);
        }
      `,
      side: THREE.DoubleSide,
      transparent: true
    });
    
    this.mesh.material = bulletMaterial;
  }
  
  public update(delta: number): void {
    super.update(delta);
    this.trailTime += delta;
    
    // 更新相机距离，做LOD

  }
